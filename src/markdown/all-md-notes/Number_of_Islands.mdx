### Solution
```python
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        rowLen = len(grid[0])
        visited = [0 * rowLen] 
        for r in range(len(grid)):
            for c in range(rowLen):
                queue = []
                if c == "1" and visited[r][c] == 0:
                    queue.append((r, c))
                    visited[r][c] = 1 
                None
        return 0
````
